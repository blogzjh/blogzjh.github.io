    public class QuickSort3 {
    public static void main(String[] args) {

        int[] arr = new int[]{23, 5, 2, 3, 4, 6, 2345, 23, 4, 5, 12, 67, 5, 3, 6, 5, 3, 1};
        sort(arr,0 , arr.length-1);
        System.out.println(Arrays.toString(arr));
    }


    public static void sort(int[] arr, int left, int right) {
        if(right<=left){
            return;
        }
        int l = left;
        int r = right;
        int base = arr[left];

        while (l < r) {
            while (r > l && arr[r] >= base) {
                r--;
            }
            while (l < r && arr[l] <= base) {
                l++;
            }

            if(l>=r){
                break;
            }else{
                swap(arr,l , r);
            }
        }

        swap(arr,l , left);

        sort(arr,left ,l-1 );
        sort(arr, l+1, right);

    }


    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

    public static void sort(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            for (int j = i; j > 0; j--) {
                if (arr[j] < arr[j - 1]){
                    swap(arr, j, j-1);
                }
            }
        }


    }


    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

}

    public class ShellSort3 {
    public static void main(String[] args) {
        int[] arr = new int[]{23, 5, 2, 3, 4, 6, 2345, 23, 4, 5, 12, 67, 5, 3, 6, 5, 3, 1};

        sort(arr);
        System.out.println(Arrays.toString(arr));
    }


    public static void sort(int[] arr) {

        int gap = arr.length;

        while (gap >= 1) {
            gap /= 2;

            for (int i = gap; i < arr.length; i++) {
                for (int j = i; j >= gap; j -= gap) {
                    if(arr[j] < arr[j-gap]){
                        swap(arr,j , j-gap);
                    }else{
                        break;
                    }
                }
            }
        }


    }

    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

    public class BubbleSort3 {
    public static void main(String[] args) {
        int[] arr = new int[]{23, 5, 2, 3, 4, 6, 2345, 23, 4, 5, 12, 67, 5, 3, 6, 5, 3, 1};
        sort(arr);
        System.out.println(Arrays.toString(arr));
    }

    public static void sort(int[] arr) {
        for (int i = arr.length - 1; i > 0; i--) {
            for (int j = 0; j < i; j++) {
                if (arr[j] > arr[j + 1]) {
                    swap(arr, j, j + 1);
                }
            }
        }
    }

    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

    堆排

    public class heapSort {
    public static void main(String[] args) {
        int[] arr = new int[]{1, 2, 4, 5, 1, 2, 4, 3, 2, 54, 23452, 543, 12, 4};
        System.out.println("最初");
        System.out.println(Arrays.toString(arr));
        System.out.println("生成大顶堆");

        heapMaker(arr);//生成大顶堆

        System.out.println(Arrays.toString(arr));

        System.out.println("循环固定，排序后");
        int size = arr.length;
        while (size > 1) {
            //1,固定最大值
            swap(arr, 0, size - 1);
            size--;
            //2,下沉：修正大顶堆
            sink(arr, 0, size);
        }


        System.out.println(Arrays.toString(arr));
    }

    //交换
    public static void swap(int[] arr, int i, int j) {
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    //创建堆：从arr[0]开始，每添加一个元素都和其父元素进行对比
    public static void heapMaker(int[] arr) {
        for (int i = 0; i < arr.length; i++) {
            int currentIndex = i;
            int fatherIndex = (i - 1) / 2;

            //因为此逻辑并未判断左右子节点之间的大小关系，所以在sink方法中需要进行判断
            while (arr[currentIndex] > arr[fatherIndex]) {
                //1.交换
                swap(arr, currentIndex, fatherIndex);
                //2.子索引变父索引，继续向上判断
                currentIndex = fatherIndex;
                fatherIndex = (fatherIndex - 1) / 2;
            }

        }
    }

    //大顶堆的修正(下沉算法)
    public static void sink(int[] arr, int index, int size) {//size:最大范围
        //1.左右子节点大小对比，
        int left = index * 2 + 1;
        int right = index * 2 + 2;

        int maxValueIndex = 0;//父、左、右的最大值对应的索引

        //2.right<size即可保证指针不越界
        while (left < size) {
            //2.1找出最大值的索引
            if (right < size && arr[right] > arr[left]) {
                maxValueIndex = right;
            } else {
                maxValueIndex = left;
            }


            if (arr[index] > arr[maxValueIndex]) {
                maxValueIndex = index;
            }


            //2.2判断是否大顶堆
            if (index == maxValueIndex) {
                break;
            }

            //2.3交换arr中的值
            swap(arr, index, maxValueIndex);

            //2.4下一轮
            index = maxValueIndex;//一定是左or右
            left = index * 2 + 1;
            right = index * 2 + 2;

        }
    }

}
